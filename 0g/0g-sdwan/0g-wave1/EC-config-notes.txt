# EC6 CONFIG
############################
configure terminal

! --- Underlay (WAN0) ---
system router EC6 create interface wan0 wan
system router EC6 ip 198.15.99.6 255.255.255.0 vlan 0

! --- LAN (LAN1) ---
system router EC6 create interface lan1 lan
system router EC6 ip 172.16.6.1 255.255.255.0 vlan 0

! --- Overlay tunnel to EC8 over WAN0 ---
tunnel create EC8_TUN
  local 198.15.99.6
  remote 198.15.99.8
  ipsec enable
  mtu 1400
  keepalive enable
exit

! --- Route remote LAN via the tunnel ---
ip route 172.16.8.0 255.255.255.0 tunnel EC8_TUN

end
write memory
